//#include<bits/stdc++.h>
#include"yzy.h"
using namespace std;
typedef unsigned long long ull;
int main()
{
    map<ull,intmymap;
    int n;
    cin >n;
    for (int i = 0; i < n; i++)
    {
        pair<ull, intp;
        cin >p.first;
        p.second = 1;
        if (mymap.find(p.first)!=mymap.end())
        {
            mymap[p.first]++;
        }
        else
        {
            mymap.insert(p);
        }
    }
    for (map<ull, int::iterator it = mymap.begin(); it != mymap.end(); ++it)
    {
        cout << (*it).first << " " << (*it).second << endl;
    }
    return 0;
}
